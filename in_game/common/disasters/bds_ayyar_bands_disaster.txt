bds_ayyar_bands_disaster = {
	fire_only_once = yes

	image = "gfx/interface/illustrations/event/backgrounds/devastation.dds"
	
	monthly_spawn_chance = 1

	can_start = {
		tag = BDS
		in_civil_war = no
		has_any_active_disaster = no
		has_ruler = yes
		NOT = { has_variable = had_bds_ayyar_bands_disaster }
		OR = {
			any_owned_location = {
				area = area:iraq_ajam_area
			}
			any_owned_location = { location = rey }
			any_owned_location = { location = karaj }
			any_owned_location = { location = qazvin }
			any_owned_location = { location = savojbolagh }
		}
	}

	can_end = {
		NOT = {
			any_owned_location = {
				area = area:iraq_ajam_area
				province = {
					has_province_modifier = bds_ayyar_bands
				}
			}
		}
	}

	modifier = {
		monthly_legitimacy = -0.1
		diplomatic_reputation = -1
		monthly_towards_decentralization = 0.1
	}

	on_start = {
		set_variable = bds_ayyar_bands_active
		add_country_modifier = { modifier = bds_ilkhanate_power_vacuum years = 15 mode = add }
		trigger_event_non_silently = flavor_bds.1
	}

	on_monthly = {
		random_list = {
			70 = { }
			15 = { trigger_event_silently = flavor_bds.2 }
			10 = { trigger_event_silently = flavor_bds.4 }
			5 = { trigger_event_silently = flavor_bds.5 }
		}
	}

	on_end = {
		trigger_event_non_silently = flavor_bds.3
		remove_country_modifier = bds_ilkhanate_power_vacuum
		remove_variable = bds_ayyar_bands_active
		set_variable = had_bds_ayyar_bands_disaster
	}
}
