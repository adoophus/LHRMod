

coa_def_own_ireland = {
	scope:actor ?= {
		OR = {
			any_owned_location = { region = region:ireland_region }
			any_subject_or_below = {
				any_owned_location = { region = region:ireland_region }
			}
		}
	}
}

coa_def_denmark_ownership_trigger = {

		OR = {	
			owns = location:kobenhaven
			any_subject_or_below = { owns = location:kobenhaven }
			any_union_partner = { owns = location:kobenhaven }			
		}
		OR = {
			NOT = { country_exists = c:DAN }
			any_subject_or_below = { this = c:DAN }
			any_union_partner = { this = c:DAN }
		}
	}
}

coa_def_spanish_cross_trigger = {
	c:SPA ?= {
		OR = {
			country_rank = country_rank:rank_kingdom
			country_rank = country_rank:rank_empire
		}
		any_subject_or_below = {
			is_subject_type = colonial_nation
			capital ?= { continent = continent:america }	
		}	
	}
}
